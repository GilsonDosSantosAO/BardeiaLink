<div class="busca-interna">
    <input type="number" id="busca" placeholder="ID">
    <button onclick="irParaElemento()">Ir</button>
</div>

<script>
    function irParaElemento() {
        const id = document.getElementById('busca').value.trim();
        const elemento = document.getElementById(id);
        if (elemento) {
            elemento.scrollIntoView({ behavior: 'smooth' });
        } else {
            alert('Elemento não encontrado!');
        }
    }

    // Detecta "Enter" no campo de busca
    document.getElementById('busca').addEventListener('keydown', function(event) {
        if (event.key === 'Enter') {
            irParaElemento();
        }
    });
</script>
